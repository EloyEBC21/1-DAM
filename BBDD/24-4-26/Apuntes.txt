INSER INTO EMPLE (EMP_NO,APELLIDO,OFICIO,DIR,FECHA_ALTA,SALARIO,COMISION, DEPT_NO)
SELECT EMP_NO,APELLIDO,OFICIO,DIR,FECHA_ALTA,SALARIO,COMISION, DEPT_NO FROM EMPLE30
WHERE DEPT_NO = 30;

De esta manera es para rellenar todas las columnas de un dato:
INSERT INTO EMPLE30 SELECT * FROM EMPLE WHERE DEPR_NO = 30;
